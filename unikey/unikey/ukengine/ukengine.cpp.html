300int tripleVowelCompare(const void *p1, const void *p2)
301{
302 VSeqPair *t1 = (VSeqPair *)p1;
303 VSeqPair *t2 = (VSeqPair *)p2;
304
305 for (int i=0; i<3; i++) {
306 if (t1->v[i] < t2->v[i])
307 return -1;
308 if (t1->v[i] > t2->v[i])
309 return 1;
310 }
311 return 0;
312}

330int VCPairCompare(const void *p1, const void *p2)
331{

334

347//----------------------------------------------------------
348bool isValidCV(ConSeq c, VowelSeq v)
349{
350 if (c == cs_nil || v == vs_nil)
351 return true;
352
353 VowelSeqInfo & vInfo = VSeqList[v];
354
355 if ((c == cs_gi && vInfo.v[0] == vnl_i) ||
356 (c == cs_qu && vInfo.v[0] == vnl_u))
357 return false; // gi doesn't go with i, qu doesn't go with u
358
359 if (c == cs_k) {
360 // k can only go with the following vowel sequences
361 static VowelSeq kVseq[] = {vs_e, vs_i, vs_y, vs_er, vs_eo, vs_eu,
362 vs_eru, vs_ia, vs_ie, vs_ier, vs_ieu, vs_ieru, vs_nil};
363 int i;
364 for (i=0; kVseq[i] != vs_nil && kVseq[i] != v; i++);
365 return (kVseq[i] != vs_nil);
366 }
367
368 //More checks
369 return true;
370}

372//----------------------------------------------------------
373bool isValidVC(VowelSeq v, ConSeq c)
374{
375 if (v == vs_nil || c == cs_nil)
376 return true;
377
378 VowelSeqInfo & vInfo = VSeqList[v];
379 if (!vInfo.conSuffix)
380 return false;
381
382 ConSeqInfo & cInfo = CSeqList[c];
383 if (!cInfo.suffix)
384 return false;
385
386 VCPair p;
387 p.v = v;
388 p.c = c;
389 if (bsearch(&p, VCPairList, VCPairCount, sizeof(VCPair), VCPairCompare))
390 return true;
391
392 return false;
393}

395//----------------------------------------------------------
396bool isValidCVC(ConSeq c1, VowelSeq v, ConSeq c2)
397{
398 if (v == vs_nil)
399 return (c1 == cs_nil || c2 != cs_nil);
400
401 if (c1 == cs_nil)
402 return isValidVC(v, c2);
403
404 if (c2 == cs_nil)
405 return isValidCV(c1, v);
406
407 bool okCV = isValidCV(c1, v);
408 bool okVC = isValidVC(v, c2);
409
410 if (okCV && okVC)
411 return true;
412
413 if (!okVC) {
414 //check some exceptions: vc fails but cvc passes
415
416 // quyn, quynh
417 if (c1 == cs_qu && v == vs_y && (c2 == cs_n || c2 == cs_nh))
418 return true;
419
420 // gieng, gie^ng
421 if (c1 == cs_gi && (v == vs_e || v == vs_er) && (c2 == cs_n || c2 == cs_ng))
422 return true;
423 }
424 return false;
425}

500//------------------------------------------------------------------
501int UkEngine::processRoof(UkKeyEvent & ev)
502{
503 if (!m_pCtrl->vietKey || m_current < 0 || m_buffer[m_current].vOffset < 0)
504 return processAppend(ev);
505
506 VnLexiName target;
507 switch (ev.evType) {
508 case vneRoof_a:
509 target = vnl_ar;
510 break;
511 case vneRoof_e:
512 target = vnl_er;
513 break;
514 case vneRoof_o:
515 target = vnl_or;
516 break;
517 default:
518 target = vnl_nonVnChar;
519 }
520
521
522 VowelSeq vs, newVs;
523 int i, vStart, vEnd;
524 int curTonePos, newTonePos, tone;
525 int changePos;
526 bool roofRemoved = false;
527
528 vEnd = m_current - m_buffer[m_current].vOffset;
529 vs = m_buffer[vEnd].vseq;
530 vStart = vEnd - (VSeqList[vs].len - 1);
531 curTonePos = vStart + getTonePosition(vs, vEnd == m_current);
532 tone = m_buffer[curTonePos].tone;
533
534 bool doubleChangeUO = false;
535 if (vs == vs_uho || vs == vs_uhoh || vs == vs_uhoi || vs == vs_uhohi) {
536 //special cases: u+o+ -> uo^, u+o -> uo^, u+o+i -> uo^i, u+oi -> uo^i
537 newVs = lookupVSeq(vnl_u, vnl_or, VSeqList[vs].v[2]);
538 doubleChangeUO = true;
539 }
540 else {
541 newVs = VSeqList[vs].withRoof;
542 }
543
544 VowelSeqInfo *pInfo;
545
546 if (newVs == vs_nil) {
547 if (VSeqList[vs].roofPos == -1)
548 return processAppend(ev); //roof is not applicable
549
550 //a roof already exists -> undo roof
551 VnLexiName curCh = m_buffer[vStart + VSeqList[vs].roofPos].vnSym;
552 if (target != vnl_nonVnChar && curCh != target)
553 return processAppend(ev); //specific roof and the roof character don't match
554
555 VnLexiName newCh = (curCh == vnl_ar)? vnl_a : ((curCh == vnl_er)? vnl_e : vnl_o);
556 changePos = vStart + VSeqList[vs].roofPos;
557
558 if (!m_pCtrl->options.freeMarking && changePos != m_current)
559 return processAppend(ev);
560
561 markChange(changePos);
562 m_buffer[changePos].vnSym = newCh;
563
564 if (VSeqList[vs].len == 3)
565 newVs = lookupVSeq(m_buffer[vStart].vnSym, m_buffer[vStart+1].vnSym, m_buffer[vStart+2].vnSym);
566 else if (VSeqList[vs].len == 2)
567 newVs = lookupVSeq(m_buffer[vStart].vnSym, m_buffer[vStart+1].vnSym);
568 else
569 newVs = lookupVSeq(m_buffer[vStart].vnSym);
570
571 pInfo = &VSeqList[newVs];
572 roofRemoved = true;
573 }
574 else {
575 pInfo = &VSeqList[newVs];
576 if (target != vnl_nonVnChar && pInfo->v[pInfo->roofPos] != target)
577 return processAppend(ev);
578
579 //check validity of new VC and CV
580 bool valid = true;
581 ConSeq c1 = cs_nil;
582 ConSeq c2 = cs_nil;
583 if (m_buffer[m_current].c1Offset != -1)
584 c1 = m_buffer[m_current-m_buffer[m_current].c1Offset].cseq;
585
586 if (m_buffer[m_current].c2Offset != -1)
587 c2 = m_buffer[m_current-m_buffer[m_current].c2Offset].cseq;
588
589 valid = isValidCVC(c1, newVs, c2);
590 if (!valid)
591 return processAppend(ev);
592
593 if (doubleChangeUO) {
594 changePos = vStart;
595 }
596 else {
597 changePos = vStart + pInfo->roofPos;
598 }
599 if (!m_pCtrl->options.freeMarking && changePos != m_current)
600 return processAppend(ev);
601 markChange(changePos);
602 if (doubleChangeUO) {
603 m_buffer[vStart].vnSym = vnl_u;
604 m_buffer[vStart+1].vnSym = vnl_or;
605 }
606 else {
607 m_buffer[changePos].vnSym = pInfo->v[pInfo->roofPos];
608 }
609 }
610
611 for (i=0; i < pInfo->len; i++) { //update sub-sequences
612 m_buffer[vStart+i].vseq = pInfo->sub[i];
613 }
614
615 //check if tone re-position is needed
616 newTonePos = vStart + getTonePosition(newVs, vEnd == m_current);
617 /* //For now, users don't seem to like the following processing, thus commented out
618 if (roofRemoved && tone != 0 &&
619 (!pInfo->complete || changePos == curTonePos)) {
620 //remove tone if the vowel sequence becomes incomplete as a result of roof removal OR
621 //if removed roof is at the same position as the current tone
622 markChange(curTonePos);
623 m_buffer[curTonePos].tone = 0;
624 } else
625 */
626 if (curTonePos != newTonePos && tone != 0) {
627 markChange(newTonePos);
628 m_buffer[newTonePos].tone = tone;
629 markChange(curTonePos);
630 m_buffer[curTonePos].tone = 0;
631 }
632
633 if (roofRemoved) {
634 m_singleMode = false;
635 processAppend(ev);
636 m_reverted = true;
637 }
638
639 return 1;
640}

945//----------------------------------------------------------
946int UkEngine::getTonePosition(VowelSeq vs, bool terminated)
947{
948 VowelSeqInfo & info = VSeqList[vs];
949 if (info.len == 1)
950 return 0;
951
952 if (info.roofPos != -1)
953 return info.roofPos;
954 if (info.hookPos != -1) {
955 if (vs == vs_uhoh || vs == vs_uhohi || vs == vs_uhohu) //u+o+, u+o+u, u+o+i
956 return 1;
957 return info.hookPos;
958 }
959
960 if (info.len == 3)
961 return 1;

1021 processAppend(ev);
1022 m_reverted = true;

1236int UkEngine::checkEscapeVIQR(UkKeyEvent & ev)
1237{

1239 return 0;

1242 if (entry.form == vnw_v || entry.form == vnw_cv) {

1252 break;

1254 case '`':
1255 case '?':
1256 case '~':
1257 case '.':

1260 }
1261 }
1262 else if (entry.form == vnw_nonVn) {

1268 case '(':
1269 escape = (ch == 'A');
1270 break;
1271 case '+':
1272 escape = (ch == 'O' || ch == 'U');
1273 break;
1274 case '\'':
1275 case '`':
1276 case '?':
1277 case '~':
1278 case '.':
1279 escape = (ch == 'A' || ch == 'E' || ch == 'I' ||
1280 ch == 'O' || ch == 'U' || ch == 'Y');

1282 }
1283 }
1284
1285 if (escape) {

1296 p->c1Offset = p->c2Offset = p->vOffset = -1;
1297 p->keyCode = ev.keyCode;
1298 p->vnSym = vnl_nonVnChar;
1299
1300 //write output
1301 m_pOutBuf[0] = '\\';
1302 m_pOutBuf[1] = ev.keyCode;
1303 *m_pOutSize = 2;
1304 m_outputWritten = true;
1305 }
1306 return escape;
1307}

1318 return 1;

1363//----------------------------------------------------------
1364int UkEngine::appendVowel(UkKeyEvent & ev)
1365{
1366 bool autoCompleted = false;
1367
1368 m_current++;
1369 WordInfo & entry = m_buffer[m_current];
1370
1371 VnLexiName lowerSym = vnToLower(ev.vnSym);
1372 VnLexiName canSym = (VnLexiName)StdVnNoTone[lowerSym];
1373
1374 entry.vnSym = canSym;
1375 entry.caps = (lowerSym != ev.vnSym);
1376 entry.tone = (lowerSym - canSym)/2;
1377 entry.keyCode = ev.keyCode;
1378
1379 if (m_current == 0 || !m_pCtrl->vietKey) {
1380 entry.form = vnw_v;
1381 entry.c1Offset = entry.c2Offset = -1;
1382 entry.vOffset = 0;
1383 entry.vseq = lookupVSeq(canSym);
1384
1385 if (!m_pCtrl->vietKey ||
1386 ((m_pCtrl->charsetId != CONV_CHARSET_UNI_CSTRING) && isalpha(entry.keyCode)) ) {
1387 return 0;
1388 }
1389 markChange(m_current);
1390 return 1;
1391 }
1392
1393 WordInfo & prev = m_buffer[m_current-1];
1394 VowelSeq vs, newVs;
1395 ConSeq cs;
1396 int prevTonePos;
1397 int tone, newTone, tonePos, newTonePos;
1398
1399 switch (prev.form) {
1400
1401 case vnw_empty:
1402 entry.form = vnw_v;
1403 entry.c1Offset = entry.c2Offset = -1;
1404 entry.vOffset = 0;
1405 entry.vseq = newVs = lookupVSeq(canSym);
1406 break;
1407
1408 case vnw_nonVn:
1409 case vnw_cvc:
1410 case vnw_vc:
1411 entry.form = vnw_nonVn;
1412 entry.c1Offset = entry.c2Offset = entry.vOffset = -1;
1413 break;
1414
1415 case vnw_v:
1416 case vnw_cv:
1417 vs = prev.vseq;
1418 prevTonePos = (m_current - 1) - (VSeqList[vs].len - 1) + getTonePosition(vs, true);
1419 tone = m_buffer[prevTonePos].tone;
1420
1421 if (lowerSym != canSym && tone != 0) //new sym has a tone, but there's is already a preceeding tone
1422 newVs = vs_nil;
1423 else {
1424 if (VSeqList[vs].len == 3)
1425 newVs = vs_nil;
1426 else if (VSeqList[vs].len == 2)
1427 newVs = lookupVSeq(VSeqList[vs].v[0], VSeqList[vs].v[1], canSym);
1428 else
1429 newVs = lookupVSeq(VSeqList[vs].v[0], canSym);
1430 }
1431
1432 if (newVs != vs_nil && prev.form == vnw_cv) {
1433 cs = m_buffer[m_current - 1 - prev.c1Offset].cseq;
1434 if (!isValidCV(cs, newVs))
1435 newVs = vs_nil;
1436 }
1437
1438 if (newVs == vs_nil) {
1439 entry.form = vnw_nonVn;
1440 entry.c1Offset = entry.c2Offset = entry.vOffset = -1;
1441 break;
1442 }
1443
1444 entry.form = prev.form;
1445 if (prev.form == vnw_cv)
1446 entry.c1Offset = prev.c1Offset + 1;
1447 else
1448 entry.c1Offset = -1;
1449 entry.c2Offset = -1;
1450 entry.vOffset = 0;
1451 entry.vseq = newVs;
1452 entry.tone = 0;
1453
1454 newTone = (lowerSym - canSym)/2;
1455 if (tone == 0) {
1456 if (newTone != 0) {
1457 tone = newTone;
1458 tonePos = getTonePosition(newVs, true) + ((m_current - 1) - VSeqList[vs].len + 1);
1459 markChange(tonePos);
1460 m_buffer[tonePos].tone = tone;
1461 return 1;
1462 }
1463 }
1464 else {
1465 newTonePos = getTonePosition(newVs, true) + ((m_current - 1) - VSeqList[vs].len + 1);
1466 if (newTonePos != prevTonePos) {
1467 markChange(prevTonePos);
1468 m_buffer[prevTonePos].tone = 0;
1469 markChange(newTonePos);
1470 if (newTone != 0)
1471 tone = newTone;
1472 m_buffer[newTonePos].tone = tone;
1473 return 1;
1474 }
1475 if (newTone != 0 && newTone != tone) {
1476 tone = newTone;
1477 markChange(prevTonePos);
1478 m_buffer[prevTonePos].tone = tone;
1479 return 1;
1480 }
1481
1482 }
1483
1484 break;
1485 case vnw_c:
1486 newVs = lookupVSeq(canSym);
1487 cs = prev.cseq;
1488 if (!isValidCV(cs, newVs)) {
1489 entry.form = vnw_nonVn;
1490 entry.c1Offset = entry.c2Offset = entry.vOffset = -1;
1491 break;
1492 }
1493
1494 entry.form = vnw_cv;
1495 entry.c1Offset = 1;
1496 entry.c2Offset = -1;

1498 entry.vseq = newVs;
1499

1501 if (entry.tone == 0)
1502 entry.tone = prev.tone;
1503 markChange(m_current - 1);
1504 prev.tone = 0;
1505 return 1;
1506 }
1507
1508 break;
1509 }
1510
1511 if (!autoCompleted &&
1512 (m_pCtrl->charsetId != CONV_CHARSET_UNI_CSTRING) &&
1513 isalpha(entry.keyCode)) {
1514 return 0;
1515 }
1516
1517 markChange(m_current);
1518 return 1;
1519}

1646 if (CSeqList[cs].len == 3)
1647 newCs = cs_nil;
1648 else if (CSeqList[cs].len == 2)

1655 c1 = cs_nil;
1656 if (prev.c1Offset != -1)
1657 c1 = m_buffer[m_current-1-prev.c1Offset].cseq;
1658
1659 int vIdx = (m_current - 1) - prev.vOffset;
1660 vs = m_buffer[vIdx].vseq;
1661 isValid = isValidCVC(c1, vs, newCs);
1662
1663 if (!isValid)
1664 newCs = cs_nil;
1665 }
1666
1667 if (newCs == cs_nil) {
1668 entry.form = vnw_nonVn;
1669 entry.c1Offset = entry.c2Offset = entry.vOffset = -1;
1670 }
1671 else {
1672 if (prev.form == vnw_c) {
1673 entry.form = vnw_c;
1674 entry.c1Offset = 0;
1675 entry.c2Offset = -1;
1676 entry.vOffset = -1;
1677 }
1678 else if (prev.form == vnw_vc) {
1679 entry.form = vnw_vc;
1680 entry.c1Offset = -1;
1681 entry.c2Offset = 0;
1682 entry.vOffset = prev.vOffset + 1;
1683 }
1684 else { //vnw_cvc
1685 entry.form = vnw_cvc;
1686 entry.c1Offset = prev.c1Offset + 1;
1687 entry.c2Offset = 0;
1688 entry.vOffset = prev.vOffset + 1;
1689 }
1690 entry.cseq = newCs;
1691 }
1692 if (m_pCtrl->charsetId != CONV_CHARSET_UNI_CSTRING)
1693 return 0;
1694 markChange(m_current);
1695 return 1;
1696 }
1697
1698 if (m_pCtrl->charsetId != CONV_CHARSET_UNI_CSTRING)
1699 return 0;
1700 markChange(m_current);
1701 return 1;
