76SingleByteCharset::SingleByteCharset(unsigned char * vnChars)
77{
78 int i;
79 m_vnChars = vnChars;
80 memset(m_stdMap, 0, 256*sizeof(UKWORD));
81 for (i=0; i<TOTAL_VNCHARS; i++) {
82 if (vnChars[i] != 0 && (i==TOTAL_VNCHARS-1 || vnChars[i] != vnChars[i+1]))
83 m_stdMap[vnChars[i]] = i + 1;
84 }

102//-------------------------------------------
103int SingleByteCharset::putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen)
104{
105 int ret;
106 unsigned char ch;
107 if (stdChar >= VnStdCharOffset) {
108 outLen = 1;
109 ch = m_vnChars[stdChar - VnStdCharOffset];
110 if (ch == 0)
111 ch = (stdChar == StdStartQuote)? PadStartQuote :
112 ((stdChar == StdEndQuote)? PadEndQuote :
113 ((stdChar == StdEllipsis)? PadEllipsis: PadChar) );
114 ret = os.putB(ch);
115 }
116 else {
117 if (stdChar > 255 || m_stdMap[stdChar]) {
118 //this character is missing in the charset
119 // output padding character
120 outLen = 1;
121 ret = os.putB(PadChar);
122 }
123 else {
124 outLen = 1;
125 ret = os.putB((UKBYTE)stdChar);
126 }
127 }
128 return ret;
129}

132int wideCharCompare(const void *ele1, const void *ele2)
133{
134 UKWORD ch1 = LOWORD(*((UKDWORD *)ele1));
135 UKWORD ch2 = LOWORD(*((UKDWORD *)ele2));
136 return (ch1 == ch2)? 0 : ((ch1 > ch2)? 1 : -1);

192UnicodeCompCharset::UnicodeCompCharset(UnicodeChar *uniChars, UKDWORD *uniCompChars)
193{
194 int i,k;
195 m_uniCompChars = uniCompChars;
196 m_totalChars = 0;
197 for (i=0; i<TOTAL_VNCHARS; i++) {
198 m_info[i].compChar = uniCompChars[i];
199 m_info[i].stdIndex = i;
200 m_totalChars++;
201 }
202
203 for (k=0, i=TOTAL_VNCHARS; k<TOTAL_VNCHARS; k++)
204 if (uniChars[k] != uniCompChars[k]) {
205 m_info[i].compChar = uniChars[k];
206 m_info[i].stdIndex = k;
207 m_totalChars++;
208 i++;
209 }
210
211 qsort(m_info, m_totalChars, sizeof(UniCompCharInfo), uniCompInfoCompare);
212}

214//---------------------------------------------
215int UnicodeCompCharset::nextInput(ByteInStream & is, StdVnChar & stdChar, int & bytesRead)
216{
217 // read first char
218
219 UniCompCharInfo key;
220 UKWORD w;
221 if (!is.getNextW(w)) {
222 bytesRead = 0;
223 return 0;
224 }
225 key.compChar = w;
226 bytesRead = 2;
227
228 UniCompCharInfo *pInfo = (UniCompCharInfo *)bsearch(&key, m_info, m_totalChars,
229 sizeof(UniCompCharInfo), uniCompInfoCompare);
230 if (!pInfo)
231 stdChar = key.compChar;
232 else {
233 stdChar = pInfo->stdIndex + VnStdCharOffset;
234 if (is.peekNextW(w)) {
235 UKDWORD hi = w;
236 if (hi > 0) {
237 key.compChar += hi << 16;
238 pInfo = (UniCompCharInfo *)bsearch(&key, m_info, m_totalChars,
239 sizeof(UniCompCharInfo), uniCompInfoCompare);
240 if (pInfo) {
241 stdChar = pInfo->stdIndex + VnStdCharOffset;
242 bytesRead += 2;
243 is.getNextW(w);
244 }
245 }
246 }
247 }
248 return 1;
249}

252int UnicodeCompCharset::putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen)
253{
254 int ret;
255 if (stdChar >= VnStdCharOffset) {
256 UKDWORD uniCompCh = m_uniCompChars[stdChar-VnStdCharOffset];
257 UKWORD lo = LOWORD(uniCompCh);
258 UKWORD hi = HIWORD(uniCompCh);
259 outLen = 2;
260 ret = os.putW(lo);
261 if (hi > 0) {
262 outLen += 2;
263 ret = os.putW(hi);
264 }
265 }
266 else {
267 outLen = 2;
268 ret = os.putW((UKWORD)stdChar);
269 }
270 return ret;
271}

280// Unicode UTF-8 //
281////////////////////////////////
282int UnicodeUTF8Charset::nextInput(ByteInStream & is, StdVnChar & stdChar, int & bytesRead)
283{
284 UKWORD w1, w2, w3;
285 UKBYTE first, second, third;
286 UnicodeChar uniCh;
287
288 bytesRead = 0;
289 if (!is.getNext(first))
290 return 0;
291 bytesRead = 1;
292
293 if (first < 0x80)
294 uniCh = first; // 1-byte sequence
295 else if ((first & 0xE0) == 0xC0) {
296 //2-byte sequence
297 if (!is.peekNext(second))
298 return 0;
299 if ((second & 0xC0) != 0x80) {
300 stdChar = INVALID_STD_CHAR;
301 return 1;
302 }
303 is.getNext(second);
304 bytesRead = 2;
305 w1 = first;
306 w2 = second;
307 uniCh = ((w1 & 0x001F) << 6) | (w2 & 0x3F);
308 }
309 else if ((first & 0xF0) == 0xE0) {
310 //3-byte sequence
311 if (!is.peekNext(second))
312 return 0;
313 if ((second & 0xC0) != 0x80) {
314 stdChar = INVALID_STD_CHAR;
315 return 1;
316 }
317 is.getNext(second);
318 bytesRead = 2;
319 if (!is.peekNext(third))
320 return 0;
321 if ((third & 0xC0) != 0x80) {
322 stdChar = INVALID_STD_CHAR;
323 return 1;
324 }
325 is.getNext(third);
326 bytesRead = 3;
327 w1 = first;
328 w2 = second;
329 w3 = third;
330 uniCh = ((w1 & 0x000F) << 12) | ((w2 & 0x003F) << 6) | (w3 & 0x003F);
331 }
332 else {
333 stdChar = INVALID_STD_CHAR;
334 return 1;
335 }
336
337 // translate to StdVnChar
338 UKDWORD key = uniCh;
339 UKDWORD *pChar = (UKDWORD *)bsearch(&key, m_vnChars, TOTAL_VNCHARS, sizeof(UKDWORD), wideCharCompare);
340 if (pChar)
341 stdChar = VnStdCharOffset + HIWORD(*pChar);
342 else stdChar = uniCh;
343 return 1;
344}

346//-------------------------------------------
347int UnicodeUTF8Charset::putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen)
348{
349 UnicodeChar uChar = (stdChar < VnStdCharOffset)?
350 (UnicodeChar)stdChar : m_toUnicode[stdChar-VnStdCharOffset];
351 int ret;
352 if (uChar < 0x0080) {
353 outLen = 1;
354 ret = os.putB((UKBYTE)uChar);
355 } else if (uChar < 0x0800) {
356 outLen = 2;
357 os.putB(0xC0 | (UKBYTE)(uChar >> 6));
358 ret = os.putB(0x80 | (UKBYTE)(uChar & 0x003F));
359 } else {
360 outLen = 3;
361 os.putB(0xE0 | (UKBYTE)(uChar >> 12));
362 os.putB(0x80 | (UKBYTE)((uChar >> 6) & 0x003F));
363 ret = os.putB(0x80 | (UKBYTE)(uChar & 0x003F));
364 }
365 return ret;
366}

371int hexDigitValue(unsigned char digit)
372{
373 if (digit >= 'a' && digit <= 'f')
374 return digit-'a'+10;
375 if (digit >= 'A' && digit <= 'F')
376 return digit-'A'+10;
377 if (digit >= '0' && digit <= '9')
378 return digit-'0';
379 return 0;
380}

383//--------------------------------------
384int UnicodeRefCharset::nextInput(ByteInStream & is, StdVnChar & stdChar, int & bytesRead)
385{
386 unsigned char ch;
387 UnicodeChar uniCh;
388 bytesRead = 0;
389 if (!is.getNext(ch))
390 return 0;
391 bytesRead = 1;
392 uniCh = ch;
393 if (ch == '&') {
394 if (is.peekNext(ch) && ch == '#') {
395 is.getNext(ch);
396 bytesRead++;
397 if (!is.eos()) {
398 is.peekNext(ch);
399 if (ch != 'x' && ch != 'X') {
400 UKWORD code = 0;
401 int digits = 0;
402 while (is.peekNext(ch) && isdigit(ch) && digits < 5) {
403 is.getNext(ch);
404 bytesRead++;
405 code = code*10 + (ch - '0');
406 digits++;
407 }
408 if (is.peekNext(ch) && ch == ';') {
409 is.getNext(ch);
410 bytesRead++;
411 uniCh = code;
412 }
413 }
414 else {
415 is.getNext(ch);
416 bytesRead++;
417 UKWORD code = 0;
418 int digits = 0;
419 while (is.peekNext(ch) && isxdigit(ch) && digits < 4) {
420 is.getNext(ch);
421 bytesRead++;
422 code = (code << 4) + hexDigitValue(ch);
423 digits++;
424 }
425 if (is.peekNext(ch) && ch == ';') {
426 is.getNext(ch);
427 bytesRead++;
428 uniCh = code;
429 }
430 } // hex digits
431 }
432 }
433 }
434
435 // translate to StdVnChar
436 UKDWORD key = uniCh;
437 UKDWORD *pChar = (UKDWORD *)bsearch(&key, m_vnChars, TOTAL_VNCHARS, sizeof(UKDWORD), wideCharCompare);
438 if (pChar)
439 stdChar = VnStdCharOffset + HIWORD(*pChar);
440 else stdChar = uniCh;
441 return 1;
442}

445//--------------------------------
446int UnicodeRefCharset::putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen)
447{
448 UnicodeChar uChar = (stdChar < VnStdCharOffset)?
449 (UnicodeChar)stdChar : m_toUnicode[stdChar-VnStdCharOffset];
450 int ret;
451 if (uChar < 128) {
452 outLen = 1;
453 ret = os.putB((UKBYTE)uChar);
454 }
455 else {
456 outLen = 2;
457 os.putB((UKBYTE)'&');
458 os.putB((UKBYTE)'#');
459
460 int i, digit, prev, base;
461 prev = 0;
462 base = 10000;
463 for (i=0; i < 5; i++) {
464 digit = uChar / base;
465 if (digit || prev) {
466 prev = 1;
467 outLen++;
468 os.putB('0' + (unsigned char)digit);
469 }
470 uChar %= base;
471 base /= 10;
472 }
473 ret = os.putB((UKBYTE)';');
474 outLen++;
475 }
476 return ret;
477}

479#define HEX_DIGIT(x) ((x < 10)? ('0'+x) : ('A'+x-10))

481//--------------------------------
482int UnicodeHexCharset::putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen)
483{
484 UnicodeChar uChar = (stdChar < VnStdCharOffset)?
485 (UnicodeChar)stdChar : m_toUnicode[stdChar-VnStdCharOffset];
486 int ret;
487 if (uChar < 256) {
488 outLen = 1;
489 ret = os.putB((UKBYTE)uChar);
490 }
491 else {
492 outLen = 3;
493 os.putB('&');
494 os.putB('#');
495 os.putB('x');
496
497 int i, digit;
498 int prev = 0;
499 int shifts = 12;
500
501 for (i=0; i < 4; i++) {
502 digit = ((uChar >> shifts) & 0x000F);
503 if (digit > 0 || prev) {
504 prev = 1;
505 outLen++;
506 os.putB((UKBYTE)HEX_DIGIT(digit));
507 }
508 shifts -= 4;
509 }
510 ret = os.putB(';');
511 outLen++;
512 }
513 return ret;
514}

597DoubleByteCharset::DoubleByteCharset(UKWORD *vnChars)
598{
599 m_toDoubleChar = vnChars;
600 memset(m_stdMap, 0, 256*sizeof(UKWORD));
601 for (int i=0; i<TOTAL_VNCHARS; i++) {
602 if (vnChars[i] >> 8) // a 2-byte character
603 m_stdMap[vnChars[i] >> 8] = 0xFFFF; //INVALID_STD_CHAR;
604 else if (m_stdMap[vnChars[i]] == 0)
605 m_stdMap[vnChars[i]] = i+1;
606 m_vnChars[i] = (i << 16) + vnChars[i]; // high word is used for StdChar index
607 }
608 qsort(m_vnChars, TOTAL_VNCHARS, sizeof(UKDWORD), wideCharCompare);
609}

611//---------------------------------------------
612int DoubleByteCharset::nextInput(ByteInStream & is, StdVnChar & stdChar, int & bytesRead)
613{
614 unsigned char ch;
615
616 // read first byte
617 bytesRead = 0;
618 if (!is.getNext(ch))
619 return 0;
620 bytesRead = 1;
621 stdChar = m_stdMap[ch];
622 if (stdChar == 0)
623 stdChar = ch;
624 else if (stdChar == 0xFFFF)
625 stdChar = INVALID_STD_CHAR;
626 else {
627 stdChar += VnStdCharOffset - 1;
628 UKBYTE hi;
629 if (is.peekNext(hi) && hi > 0) {
630 //test if a double-byte character is encountered
631 UKDWORD key = MAKEWORD(ch,hi);
632 UKDWORD *pChar = (UKDWORD *)bsearch(&key, m_vnChars, TOTAL_VNCHARS, sizeof(UKDWORD), wideCharCompare);
633 if (pChar) {
634 stdChar = VnStdCharOffset + HIWORD(*pChar);
635 bytesRead = 2;
636 is.getNext(hi);
637 }
638 }
639 }
640 return 1;
641}

644int DoubleByteCharset::putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen)

651 outLen = 2;

656 unsigned char b = (unsigned char)wCh;
657 if (m_stdMap[b] == 0xFFFF)
658 b = PadChar;

664 ret = os.putB((UKBYTE)(wCh & 0x00FF));
665 if (wCh & 0xFF00) {

668 }
669*/
670 }
671 else {
672 if (stdChar > 255 || m_stdMap[stdChar]) {
673 outLen = 1;
674 ret = os.putB((UKBYTE)PadChar);
675 }
676 else {
677 outLen = 1;
678 ret = os.putB((UKBYTE)stdChar);
679 }
680 }
681 return ret;

731void VIQRCharset::startInput()
732{
733 m_suspicious = 0;
734 m_atWordBeginning = 1;
735 m_gotTone = 0;
736 m_escAll = 0;
737 if (VnCharsetLibObj.m_options.viqrEsc)

741//---------------------------------------------------
742int VIQRCharset::nextInput(ByteInStream & is, StdVnChar & stdChar, int & bytesRead)
743{
744 unsigned char ch1;
745 bytesRead = 0;
746
747 if (!is.getNext(ch1))
748 return 0;
749 bytesRead = 1;
750 stdChar = m_stdMap[ch1];
751
752 if (VnCharsetLibObj.m_options.viqrEsc) {
753 if (VnCharsetLibObj.m_VIQREscPatterns.foundAtNextChar(ch1)!=-1) {
754 m_escAll = 1;
755 }
756 }
757
758 if (m_escAll && (ch1==' ' || ch1=='\t' || ch1=='\r' || ch1=='\n'))
759 m_escAll = 0;
760
761 if (ch1 == '\\') {
762 // ecape character , try to read next
763 if (!is.getNext(ch1)) {
764 bytesRead++;
765 stdChar = m_stdMap[ch1];
766 }
767 }
768
769 if (stdChar < 256) {
770 stdChar = ch1;
771 }
772 else if (!m_escAll && !is.eos()) {
773 // try to read the next byte
774 unsigned char ch2;
775 is.peekNext(ch2);
776 unsigned char upper = toupper(ch1);
777 if ((!VnCharsetLibObj.m_options.smartViqr || m_atWordBeginning) &&
778 upper == 'D' && (ch2 == 'd' || ch2 == 'D'))
779 {
780 is.getNext(ch2);
781 bytesRead++;
782 stdChar += 2; // dd is 2 positions after d.
783 }
784 else {
785 StdVnChar index = m_stdMap[ch2];
786
787 int cond;
788 if (m_suspicious) {
789 cond = IS_VOWEL(ch1) &&
790 ( index == 2 || index == 4 || index == 8 || //not accepting ? . in suspicious mode
791 (index == 12 && (upper == 'A' || upper == 'E' || upper == 'O')) ||
792 (m_stdMap[ch2] == 24 && upper== 'A') ||
793 (m_stdMap[ch2] == 26 && (upper == 'O' || upper == 'U')) );
794 if (cond)
795 m_suspicious = 0;
796 }
797 else
798 cond = IS_VOWEL(ch1) &&
799 ((index <= 10 && index > 0 && (!m_gotTone || (index!=6 && index!=10)) ) ||
800 (index == 12 && (upper == 'A' || upper == 'E' || upper == 'O')) ||
801 (m_stdMap[ch2] == 24 && upper== 'A') ||
802 (m_stdMap[ch2] == 26 && (upper == 'O' || upper == 'U')) );
803
804 if (cond) {
805 if (index > 0)
806 m_gotTone = 1; //we have a tone/breve/hook in the current word
807
808 // ok, take this byte
809 is.getNext(ch2);
810 bytesRead++;
811 int offset = m_stdMap[ch2];
812 if (offset == 26) offset = 24;
813 if (offset == 24 && (ch1 == 'u' || ch1 == 'U'))
814 offset = 12;
815 stdChar += offset;
816 // check next byte
817 if (is.peekNext(ch2)) {
818 if (index > 10 && m_stdMap[ch2] > 0 && m_stdMap[ch2] <= 10) {
819 // ok, take one more byte
820 is.getNext(ch2);
821 bytesRead++;
822 stdChar += m_stdMap[ch2];
823 }
824 }
825 }
826 }
827 }
828 m_atWordBeginning = (stdChar < 256);
829 if (stdChar < 256) {
830 m_gotTone = 0; //reset this flag because we are at the beginning of a new word
831 }
832
833 // adjust stdChar
834 if (stdChar >= 256)
835 stdChar += VnStdCharOffset - 256;
836 return 1;
837}

850//---------------------------------------------------
851int VIQRCharset::putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen)
852{
853 int ret;
854 UKBYTE b;
855 if (stdChar >= VnStdCharOffset) {
856 outLen = 1;
857 UKDWORD dw = m_vnChars[stdChar-VnStdCharOffset];
858
859 unsigned char first = (unsigned char)dw;
860 unsigned char firstUpper = toupper(first);
861
862 b = (UKBYTE)dw;
863 ret = os.putB(b);
864 if (VnCharsetLibObj.m_VIQROutEscPatterns.foundAtNextChar(b) != -1)
865 m_noOutEsc = 1;
866
867 if (m_noOutEsc && (b==' ' || b=='\t' || b=='\r' || b=='\n'))
868 m_noOutEsc = 0;
869
870 if (dw & 0x0000FF00) {
871 // second byte is present
872 unsigned char second = (UKBYTE)(dw >> 8);
873 outLen++;
874 ret = os.putB(second);
875
876 if (dw & 0x00FF0000) {
877 //third byte is present
878 outLen++;
879 ret = os.putB((UKBYTE)(dw >> 16));
880 m_escapeTone = 0;
881 }
882 else {
883 UKWORD index = m_stdMap[second];
884 m_escapeTone = (index == 12 || index == 24 || index == 26);
885 }
886
887 VnCharsetLibObj.m_VIQROutEscPatterns.reset();
888
889 m_escapeBowl = 0;
890 m_escapeHook = 0;
891 m_escapeRoof = 0;
892 }
893 else {
894 m_escapeTone = IS_VOWEL(first);
895 m_escapeBowl = (firstUpper == 'A');
896 m_escapeHook = (firstUpper == 'U' || firstUpper == 'O');
897 m_escapeRoof = (firstUpper == 'A' || firstUpper == 'E' || firstUpper == 'O');
898 }
899 }
900 else {
901 if (stdChar > 255) {
902 outLen = 1;
903 ret = os.putB((UKBYTE)PadChar);
904 if (VnCharsetLibObj.m_VIQROutEscPatterns.foundAtNextChar((UKBYTE)PadChar) != -1)
905 m_noOutEsc = 1;
906 }
907 else {
908 outLen = 1;
909 UKWORD index = m_stdMap[stdChar];
910 if (!VnCharsetLibObj.m_options.viqrMixed && !m_noOutEsc &&
911 (stdChar=='\\' ||
912 (index > 0 && index <= 10 && m_escapeTone) ||
913 (index == 12 && m_escapeRoof) ||
914 (index == 24 && m_escapeBowl) ||
915 (index == 26 && m_escapeHook))) {
916 //(m_stdMap[stdChar] > 0 && m_stdMap[stdChar] <= 26)) {
917 // tone mark, needs an escape character
918 outLen++;
919 ret = os.putB('\\');
920 if (VnCharsetLibObj.m_VIQROutEscPatterns.foundAtNextChar('\\') != -1)
921 m_noOutEsc = 1;
922 }
923 b = (UKBYTE)stdChar;
924 ret = os.putB(b);
925 if (VnCharsetLibObj.m_VIQROutEscPatterns.foundAtNextChar(b) != -1)
926 m_noOutEsc = 1;
927 if (m_noOutEsc && (b==' ' || b=='\t' || b=='\r' || b=='\n'))
928 m_noOutEsc = 0;
929 }
930 // reset escape marks
931 m_escapeBowl = 0;
932 m_escapeRoof = 0;
933 m_escapeHook = 0;
934 m_escapeTone = 0;
935 }
936 return ret;
937}

1169WinCP1258Charset::WinCP1258Charset(UKWORD *compositeChars, UKWORD *precomposedChars)
1170{
1171 int i,k;
1172 m_toDoubleChar = compositeChars;
1173 memset(m_stdMap, 0, 256*sizeof(UKWORD));
1174
1175 // encode composite chars
1176 for (i=0; i<TOTAL_VNCHARS; i++) {
1177 if (compositeChars[i] >> 8) // a 2-byte character
1178 m_stdMap[compositeChars[i] >> 8] = 0xFFFF; //INVALID_STD_CHAR;
1179 else if (m_stdMap[compositeChars[i]] == 0)
1180 m_stdMap[compositeChars[i]] = i+1;
1181
1182 m_vnChars[i] = (i << 16) + compositeChars[i]; // high word is used for StdChar index
1183 }
1184
1185 m_totalChars = TOTAL_VNCHARS;
1186
1187 //add precomposed chars to the table
1188 for (k=0, i=TOTAL_VNCHARS; k<TOTAL_VNCHARS; k++)
1189 if (precomposedChars[k] != compositeChars[k]) {
1190 if (precomposedChars[k] >> 8) // a 2-byte character
1191 m_stdMap[precomposedChars[k] >> 8] = 0xFFFF; //INVALID_STD_CHAR;
1192 else if (m_stdMap[precomposedChars[k]] == 0)
1193 m_stdMap[precomposedChars[k]] = k+1;
1194
1195 m_vnChars[i] = (k << 16) + precomposedChars[k];
1196 m_totalChars++;
1197 i++;
1198 }
1199
1200 qsort(m_vnChars, m_totalChars, sizeof(UKDWORD), wideCharCompare);
1201}

1204//---------------------------------------------------------------------
1205// This fuction is basically the same as that of DoubleByteCharset
1206// with m_totalChars is used instead of constant TOTAL_VNCHARS
1207//---------------------------------------------------------------------
1208int WinCP1258Charset::nextInput(ByteInStream & is, StdVnChar & stdChar, int & bytesRead)
1209{
1210 unsigned char ch;
1211
1212 // read first byte
1213 bytesRead = 0;
1214 if (!is.getNext(ch))
1215 return 0;
1216 bytesRead = 1;
1217 stdChar = m_stdMap[ch];
1218 if (stdChar == 0)
1219 stdChar = ch;
1220 else if (stdChar == 0xFFFF)
1221 stdChar = INVALID_STD_CHAR;
1222 else {
1223 stdChar += VnStdCharOffset - 1;
1224 UKBYTE hi;
1225 if (is.peekNext(hi) && hi > 0) {
1226 //test if a double-byte character is encountered
1227 UKDWORD key = MAKEWORD(ch,hi);
1228 UKDWORD *pChar = (UKDWORD *)bsearch(&key, m_vnChars, m_totalChars, sizeof(UKDWORD), wideCharCompare);
1229 if (pChar) {
1230 stdChar = VnStdCharOffset + HIWORD(*pChar);
1231 bytesRead = 2;
1232 is.getNext(hi);
1233 }
1234 }
1235 }
1236 return 1;
1237}

1256 b = PadChar;
1257 outLen = 1;

1263 outLen = 1;
1264 ret = os.putB((UKBYTE)PadChar);
1265 }
1266 else {
1267 outLen = 1;
1268 ret = os.putB((UKBYTE)stdChar);
1269 }
1270 }
1271 return ret;
