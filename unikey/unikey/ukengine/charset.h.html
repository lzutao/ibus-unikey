87class DllInterface VnCharset {
88public:
89 virtual void startInput() {};
90 virtual void startOutput() {};
91// virtual UKBYTE *nextInput(UKBYTE *input, int inLen, StdVnChar & stdChar, int & bytesRead) = 0;
92 virtual int nextInput(ByteInStream & is, StdVnChar & stdChar, int & bytesRead) = 0;
93
94 //------------------------------------------------------------------------
95 // put a character to the output after converting it
96 // Arguments:
97 // output[in]: output buffer
98 // stdChar[in]: character in standard charset
99 // outLen[out]: length of converted sequence
100 // maxAvail[in]: max length available.
101 // Returns: next position in output
102 //------------------------------------------------------------------------
103 virtual int putChar(ByteOutStream & os, StdVnChar stdChar, int & outLen) = 0;
104 virtual int elementSize();
105 virtual ~VnCharset() {}
106};
