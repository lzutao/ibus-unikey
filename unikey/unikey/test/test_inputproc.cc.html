16START_TEST(testKeyCodeToEvent) {
17 struct _tcase {
18 UkInputMethod im;
19 uint32_t keycode;
20 UkKeyEvent event;
21 };
22 typedef struct _tcase tcase;
23
24 tcase test_cases[] = {
25 {UkTelex, 'z', {vneTone0, ukcVn, vnl_z, 'z', 0}},
26 {UkTelex, 's', {vneTone1, ukcVn, vnl_s, 's', 1}},
27 {UkTelex, 'f', {vneTone2, ukcNonVn, vnl_f, 'f', 2}},
28 {UkTelex, 'r', {vneTone3, ukcVn, vnl_r, 'r', 3}},
29 {UkTelex, 'x', {vneTone4, ukcVn, vnl_x, 'x', 4}},
30 {UkTelex, 'j', {vneTone5, ukcNonVn, vnl_j, 'j', 5}},
31
32 // mapping
33 {UkTelex, '[', {vneMapChar, ukcVn, vnl_oh, '[', 5}}, // 5 from the last test
34
35 {UkVni, '0', {vneTone0, ukcNonVn, vnl_nonVnChar, '0', 0}},
36 {UkVni, '1', {vneTone1, ukcNonVn, vnl_nonVnChar, '1', 1}},
37 {UkVni, '2', {vneTone2, ukcNonVn, vnl_nonVnChar, '2', 2}},
38 {UkVni, '3', {vneTone3, ukcNonVn, vnl_nonVnChar, '3', 3}},
39 {UkVni, '4', {vneTone4, ukcNonVn, vnl_nonVnChar, '4', 4}},
40 {UkVni, '5', {vneTone5, ukcNonVn, vnl_nonVnChar, '5', 5}},
41 };
42
43 tcase emptycase;
44 size_t num_cases = sizeof(test_cases) / sizeof(emptycase);
45
46 UkInputMethod old_im = UkTelex;
47
48 uint32_t keycode = 0;
49 UkKeyEvent *want;
50 UkKeyEvent event = {0};
51 UkInputMethod im = UkTelex;
52
53 for (size_t i = 0; i < num_cases; ++i) {
54 im = test_cases[i].im;
55 keycode = test_cases[i].keycode;
56 want = &(test_cases[i].event);
57 if (im != old_im) {
58 kProcessor.setIM(im);
59 old_im = im;
60 }
61
62 kProcessor.keyCodeToEvent(keycode, event);
63 ck_assert(compare_event(event, *want));
64 }
65}

103bool compare_event(const UkKeyEvent &event, const UkKeyEvent &other) {
104 return event.evType == other.evType && event.chType == other.chType &&
105 event.vnSym == other.vnSym && event.keyCode == other.keyCode &&
106 event.tone == other.tone;
