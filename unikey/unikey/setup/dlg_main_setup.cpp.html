18GtkWidget *unikey_main_setup_dialog_new()
19{
20 ibus_init();
21 IBusBus *bus = ibus_bus_new();
22 if (!ibus_bus_is_connected(bus))
23 {
24 g_warning("unikey_main_setup_dialog_new: Cannot connect to IBus daemon.\n");
25 exit(0);
26 }
27
28 g_signal_connect(bus, "disconnected", G_CALLBACK(gtk_main_quit), NULL);
29 config = ibus_bus_get_config(bus);
30
31 GtkBuilder *builder = gtk_builder_new();
32 gtk_builder_add_from_file(builder, UI_DATA_DIR "/setup-main.ui", NULL);
33
34 GtkDialog *dlgMain = GTK_DIALOG(gtk_builder_get_object(builder, "dlg_main_setup"));
35
36 init_dialog_controls(builder);
37
38 g_object_unref(builder);
39
40 return GTK_WIDGET(dlgMain);
41}

43void init_dialog_controls(GtkBuilder *builder)
44{
45 GtkWidget *wid;
46 guint i;
47 gchar *str;
48 gboolean b;
49 GtkListStore *ls;
50 GtkTreeIter iter;
51
52 // fill inputmethod
53 ls = GTK_LIST_STORE(gtk_builder_get_object(builder, "list_input_method"));
54 gtk_list_store_clear(ls);
55 for (i = 0; i < NUM_INPUTMETHOD; i++)
56 {
57 gtk_list_store_append(ls, &iter);
58 gtk_list_store_set(ls, &iter, 0, Unikey_IMNames[i], -1);
59 }
60
61 // fill outputcharset
62 ls = GTK_LIST_STORE(gtk_builder_get_object(builder, "list_output_charset"));
63 gtk_list_store_clear(ls);
64 for (i = 0; i < NUM_OUTPUTCHARSET; i++)
65 {
66 gtk_list_store_append(ls, &iter);
67 gtk_list_store_set(ls, &iter, 0, Unikey_OCNames[i], -1);
68 }
69
70 wid = GTK_WIDGET(gtk_builder_get_object(builder, "cbb_input_method"));
71 g_signal_connect(wid, "changed", G_CALLBACK(input_method_combo_box_changed_cb), NULL);
72 i = 0;
73 if (ibus_unikey_config_get_string(config, CONFIG_SECTION, CONFIG_INPUTMETHOD, &str))
74 {
75 for (; i < NUM_INPUTMETHOD; i++)
76 {
77 if (strcasecmp(str, Unikey_IMNames[i]) == 0)
78 break;
79 }
80 }
81 gtk_combo_box_set_active(GTK_COMBO_BOX(wid), i);
82
83 wid = GTK_WIDGET(gtk_builder_get_object(builder, "cbb_output_charset"));
84 g_signal_connect(wid, "changed", G_CALLBACK(output_charset_combo_box_changed_cb), NULL);
85 i = 0;
86 if (ibus_unikey_config_get_string(config, CONFIG_SECTION, CONFIG_OUTPUTCHARSET, &str))
87 {
88 for (; i < NUM_OUTPUTCHARSET; i++)
89 {
90 if (strcasecmp(str, Unikey_OCNames[i]) == 0)
91 break;
92 }
93 }
94 gtk_combo_box_set_active(GTK_COMBO_BOX(wid), i);
95
96 wid = GTK_WIDGET(gtk_builder_get_object(builder, "check_spellcheck"));
97 g_signal_connect(wid, "toggled", G_CALLBACK(update_config_toggle_cb), (void *)CONFIG_SPELLCHECK);
98 if (!ibus_unikey_config_get_boolean(config, CONFIG_SECTION, CONFIG_SPELLCHECK, &b))
99 b = DEFAULT_CONF_SPELLCHECK;
100 gtk_toggle_button_set_active(GTK_TOGGLE_BUTTON(wid), b);
101
102 wid = GTK_WIDGET(gtk_builder_get_object(builder, "check_autorestorenonvn"));
103 g_signal_connect(wid, "toggled", G_CALLBACK(update_config_toggle_cb), (void *)CONFIG_AUTORESTORENONVN);
104 if (!ibus_unikey_config_get_boolean(config, CONFIG_SECTION, CONFIG_AUTORESTORENONVN, &b))
105 b = DEFAULT_CONF_AUTONONVNRESTORE;
106 gtk_toggle_button_set_active(GTK_TOGGLE_BUTTON(wid), b);
107
108 wid = GTK_WIDGET(gtk_builder_get_object(builder, "check_modernstyle"));
109 g_signal_connect(wid, "toggled", G_CALLBACK(update_config_toggle_cb), (void *)CONFIG_MODERNSTYLE);
110 if (!ibus_unikey_config_get_boolean(config, CONFIG_SECTION, CONFIG_MODERNSTYLE, &b))
111 b = DEFAULT_CONF_MODERNSTYLE;
112 gtk_toggle_button_set_active(GTK_TOGGLE_BUTTON(wid), b);
113
114 wid = GTK_WIDGET(gtk_builder_get_object(builder, "check_freemarking"));
115 g_signal_connect(wid, "toggled", G_CALLBACK(update_config_toggle_cb), (void *)CONFIG_FREEMARKING);
116 if (!ibus_unikey_config_get_boolean(config, CONFIG_SECTION, CONFIG_FREEMARKING, &b))
117 b = DEFAULT_CONF_FREEMARKING;

148void macro_edit_button_cb(GtkButton *btn, gpointer user_data)
149{
150 GtkWidget *parent_dlg = GTK_WIDGET(user_data);
151
152 gtk_widget_set_sensitive(parent_dlg, FALSE);
153
154 GtkWidget *dlg = unikey_macro_dialog_new();
155
156 gchar *macrofile = get_macro_file();
157
158 CMacroTable macro;
159 macro.init();
160 macro.loadFromFile(macrofile);
161
162 unikey_macro_dialog_load_macro(GTK_DIALOG(dlg), macro);
163
164 int ret = gtk_dialog_run(GTK_DIALOG(dlg));
165
166 if (ret == GTK_RESPONSE_OK)
167 {
168 unikey_macro_dialog_save_macro(GTK_DIALOG(dlg), &macro);
169
170 GFile *f = g_file_get_parent(g_file_new_for_path(macrofile));
171 if (g_file_query_exists(f, NULL) == FALSE)
172 {
173 g_file_make_directory_with_parents(f, NULL, NULL);
174 }
175 g_object_unref(f);
176
177 macro.writeToFile(macrofile);
178 }
179
180 g_free(macrofile);
181
182 gtk_widget_destroy(dlg);
183
184 gtk_widget_set_sensitive(parent_dlg, TRUE);
185}
