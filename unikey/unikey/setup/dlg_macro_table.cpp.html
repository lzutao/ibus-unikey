72void append_macro_to_list_store(GtkListStore* list, CMacroTable macro)
73{
74 GtkTreeIter iter;
75 gchar key[MAX_MACRO_KEY_LEN*3];
76 gchar value[MAX_MACRO_TEXT_LEN*3];
77 gchar* oldkey;
78 UKBYTE* p;
79 int inLen, maxOutLen;
80 int i, ret;
81
82 for (i = 0 ; i < macro.getCount(); i++)
83 {
84 // get key and convert to XUTF charset
85 p = (UKBYTE*)macro.getKey(i);
86 inLen = -1;
87 maxOutLen = sizeof(key);
88 ret = VnConvert(CONV_CHARSET_VNSTANDARD, CONV_CHARSET_XUTF8,
89 p, (UKBYTE*)key,
90 &inLen, &maxOutLen);
91 if (ret != 0)
92 continue;
93
94 // check if any key same as newkey
95 gboolean b = gtk_tree_model_get_iter_first(GTK_TREE_MODEL(list), &iter);
96 while (b)
97 {
98 gtk_tree_model_get(GTK_TREE_MODEL(list), &iter, COL_KEY, &oldkey, -1);
99
100 if (strcasecmp(oldkey, key) == 0)
101 {
102 break;
103 }
104
105 b = gtk_tree_model_iter_next(GTK_TREE_MODEL(list), &iter);
106 }
107 if (b)
108 {
109 continue;
110 }
111 // end check
112
113 // get value and convert to XUTF charset
114 p = (UKBYTE*)macro.getText(i);
115 inLen = -1;
116 maxOutLen = sizeof(value);
117 ret = VnConvert(CONV_CHARSET_VNSTANDARD, CONV_CHARSET_XUTF8,
118 p, (UKBYTE*)value,
119 &inLen, &maxOutLen);
120 if (ret != 0)
121 continue;
122
123 // append to liststore
124 gtk_list_store_append(list, &iter);
125 gtk_list_store_set(list, &iter, COL_KEY, key, COL_VALUE, value, -1);
126 }

149void unikey_macro_dialog_save_macro(GtkDialog* dialog, CMacroTable* macro)
150{
151 GtkTreeView* tree;
152 GtkTreeModel* model;
153 GtkTreeIter iter;
154 gboolean b;
155 gchar *key, *value;
156
157 macro->resetContent();
158
159 tree = GTK_TREE_VIEW(g_object_get_data(G_OBJECT(dialog), "tree_macro"));
160 model = GTK_TREE_MODEL(gtk_tree_view_get_model(tree));
161
162 b = gtk_tree_model_get_iter_first(model, &iter);
163 while (b == TRUE)
164 {
165 gtk_tree_model_get(model, &iter, COL_KEY, &key, COL_VALUE, &value, -1);
166
167 if (strcasecmp(key, STR_NULL_ITEM) != 0)
168 {
169 macro->addItem(key, value, CONV_CHARSET_XUTF8);
170 }
171
172 b = gtk_tree_model_iter_next(model, &iter);
173 }

176void check_last_macro_in_list(GtkListStore* list)
177{
178 GtkTreeIter iter;
179 gchar *key;
180 gint n;
181
182 // get number item in list
183 n = gtk_tree_model_iter_n_children(GTK_TREE_MODEL(list), NULL);
184
185 if (n > 0)
186 {
187 // get last item
188 gtk_tree_model_iter_nth_child(GTK_TREE_MODEL(list), &iter, NULL, n-1);
189
190 // get key of item
191 gtk_tree_model_get(GTK_TREE_MODEL(list), &iter, COL_KEY, &key, -1);
192
193 // if key is value used for NULL item
194 if (strcmp(key, STR_NULL_ITEM) == 0)
195 {
196 return;
197 }
198 }
199
200 // if last item is valid item or no item in list, add new NULL item
201 gtk_list_store_append(list, &iter);
202 gtk_list_store_set(list, &iter,
203 COL_KEY, STR_NULL_ITEM,
204 COL_VALUE, STR_NULL_ITEM,
205 -1);
206}

208void key_edited_cb (GtkCellRendererText *celltext,
209 const gchar *string_path,
210 const gchar *newkey,
211 gpointer data)
212{
213 GtkTreeView *tree;
214 GtkTreeModel *model;
215 GtkTreeIter iter;
216 gchar *key, *oldkey, *oldvalue;
217 gchar nkey[MAX_MACRO_KEY_LEN];
218 gboolean b;
219
220 tree = GTK_TREE_VIEW(data);
221 model = gtk_tree_view_get_model(tree);
222
223 strncpy(nkey, newkey, MAX_MACRO_KEY_LEN-1);
224 nkey[MAX_MACRO_KEY_LEN-1] = '\0';
225
226 if (strcmp(nkey, STR_NULL_ITEM) == 0
227 || (strlen(STR_NULL_ITEM) != 0 && strlen(nkey) == 0))
228 {
229 return;
230 }
231
232 // check if any key same as newkey
233 b = gtk_tree_model_get_iter_first(model, &iter);
234 while (b)
235 {
236 gtk_tree_model_get(model, &iter, COL_KEY, &key, -1);
237 if (strcasecmp(key, nkey) == 0)
238 {
239 return;
240 }
241
242 b = gtk_tree_model_iter_next(model, &iter);
243 }
244 // end check
245
246 // get iter of newkey
247 gtk_tree_model_get_iter_from_string(model, &iter, string_path);
248 // get old value of that iter
249 gtk_tree_model_get(model, &iter, COL_KEY, &oldkey, COL_VALUE, &oldvalue, -1);
250
251 gtk_list_store_set(GTK_LIST_STORE(model), &iter, COL_KEY, nkey, -1);
252
253 if (strcmp(oldkey, STR_NULL_ITEM) == 0)
254 {
255 gtk_list_store_set(GTK_LIST_STORE(model), &iter, COL_VALUE, MACRO_DEFAULT_VALUE);
256 }
257
258 check_last_macro_in_list(GTK_LIST_STORE(model));
259}

289{
290 GtkTreeView *treeview;

293 GtkTreeIter iter;

297
298 select = gtk_tree_view_get_selection(treeview);
299
300 list = GTK_LIST_STORE(gtk_tree_view_get_model(treeview));
301
302 if (gtk_tree_selection_get_selected(select, NULL, &iter) == TRUE)

304 gtk_tree_model_get(GTK_TREE_MODEL(list), &iter, COL_KEY, &key, -1);
305
306 if (strcmp(key, STR_NULL_ITEM) != 0)

336void import_macro_clicked_cb(GtkButton *button, gpointer data)
337{
338 GtkWidget* file;
339 GtkTreeView* tree;
340 GtkListStore* list;
341 GtkTreeIter iter;
342 CMacroTable macro;
343 gchar* fn;
344 int r, n;
345
346 file = gtk_file_chooser_dialog_new(_("Choose file to import"),
347 GTK_WINDOW(data),
348 GTK_FILE_CHOOSER_ACTION_OPEN,
349 GTK_STOCK_CANCEL, GTK_RESPONSE_CANCEL,
350 GTK_STOCK_OPEN, GTK_RESPONSE_OK,
351 NULL);
352
353 gtk_widget_set_sensitive(GTK_WIDGET(data), FALSE);
354
355 r = gtk_dialog_run(GTK_DIALOG(file));
356 if (r == GTK_RESPONSE_OK)
357 {
358 fn = gtk_file_chooser_get_filename(GTK_FILE_CHOOSER(file));
359 macro.init();
360 macro.loadFromFile(fn);
361 g_free(fn);
362
363 tree = GTK_TREE_VIEW(g_object_get_data(G_OBJECT(data), "tree_macro"));
364 list = GTK_LIST_STORE(gtk_tree_view_get_model(tree));
365
366 n = gtk_tree_model_iter_n_children(GTK_TREE_MODEL(list), NULL); // get number of iter
367 gtk_tree_model_iter_nth_child(GTK_TREE_MODEL(list), &iter, NULL, n-1); // get last iter
368 gtk_list_store_remove(list, &iter); // remove last iter (...)
369
370 append_macro_to_list_store(list, macro);
371
372 check_last_macro_in_list(list); // add iter (...)
373
374 // select first iter
375 GtkTreeSelection* select = gtk_tree_view_get_selection(tree);
376 gtk_tree_model_get_iter_first(GTK_TREE_MODEL(list), &iter);
377 gtk_tree_selection_select_iter(select, &iter);
378 }
379
380 gtk_widget_destroy(file);
381
382 gtk_widget_set_sensitive(GTK_WIDGET(data), TRUE);
