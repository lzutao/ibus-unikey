21int main(int argc, char **argv)

27 // Parse the command line
28 GOptionContext *context = g_option_context_new("- ibus unikey setup component");
29 g_option_context_add_main_entries(context, entries, "ibus-unikey");
30 GError *error = NULL;
31 gboolean parse_ok = g_option_context_parse(context, &argc, &argv, &error);
32
33 if (!parse_ok)
34 {
35 g_print("Option parsing failed: %s\n", error->message);
36 g_error_free(error);
37 exit(EXIT_FAILURE);
38 }
39
40 if (kVersion)
41 {
42 g_print(PACKAGE_STRING " (setup component)"

46 "\n License: GNU GPL3"
47 "\n");
48 return 0;
49 }
50
51 gtk_init(&argc, &argv);
52
53 gtk_window_set_default_icon_from_file(PKGDATADIR "/icons/ibus-unikey.png", NULL);

55 GtkWidget* main_dlg = unikey_main_setup_dialog_new(); // create main dlg
