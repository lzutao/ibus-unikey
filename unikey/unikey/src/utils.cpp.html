76// code from x-unikey, for convert charset that not is XUtf-8
77int latinToUtf(unsigned char *dst, unsigned char *src, int inSize, int *pOutSize)
78{
79 int i;
80 int outLeft;
81 unsigned char ch;
82
83 outLeft = *pOutSize;
84
85 for (i = 0; i < inSize; i++)
86 {
87 ch = *src++;
88 if (ch < 0x80)
89 {
90 outLeft -= 1;
91 if (outLeft >= 0)
92 *dst++ = ch;
93 }
94 else
95 {
96 outLeft -= 2;
97 if (outLeft >= 0)
98 {
99 *dst++ = (0xC0 | ch >> 6);
100 *dst++ = (0x80 | (ch & 0x3F));
101 }
102 }
103 }
104
105 *pOutSize = outLeft;
106 return (outLeft >= 0);
107}

110 const gchar *section,

112 gchar **result)
113{
114 GVariant *value = ibus_config_get_value(config, section, name);
