79static void ibus_unikey_engine_class_init(IBusUnikeyEngineClass *klass)
80{
81 GObjectClass *object_class = G_OBJECT_CLASS(klass);
82 object_class->constructor = ibus_unikey_engine_constructor;
83
84 IBusObjectClass *ibus_object_class = IBUS_OBJECT_CLASS(klass);
85 ibus_object_class->destroy = (IBusObjectDestroyFunc)ibus_unikey_engine_destroy;
86
87 IBusEngineClass *engine_class = IBUS_ENGINE_CLASS(klass);
88 engine_class->process_key_event = ibus_unikey_engine_process_key_event;
89 engine_class->reset = ibus_unikey_engine_reset;
90 engine_class->enable = ibus_unikey_engine_enable;
91 engine_class->disable = ibus_unikey_engine_disable;
92 engine_class->focus_in = ibus_unikey_engine_focus_in;
93 engine_class->focus_out = ibus_unikey_engine_focus_out;
94 engine_class->property_activate = ibus_unikey_engine_property_activate;
95
96 kParentClass = (IBusEngineClass* )ibus_unikey_engine_parent_class;
97}

99static void ibus_unikey_engine_init(IBusUnikeyEngine *unikey)
100{
101 if (g_object_is_floating (unikey))
102 g_object_ref_sink (unikey); // make engine sink
103 ibus_unikey_engine_load_config(unikey);
104
105 unikey->preeditstr = new std::string();
106 ibus_unikey_engine_create_property_list(unikey);
107}

170static void ibus_unikey_engine_load_config(IBusUnikeyEngine *unikey)
171{
172 gchar *str;
173 gboolean b;
174 guint i;
175
176 // Set default options
177 unikey->im = Unikey_IM[0];
178 unikey->oc = Unikey_OC[0];
179 unikey->ukopt.spellCheckEnabled = DEFAULT_CONF_SPELLCHECK;
180 unikey->ukopt.autoNonVnRestore = DEFAULT_CONF_AUTONONVNRESTORE;
181 unikey->ukopt.modernStyle = DEFAULT_CONF_MODERNSTYLE;
182 unikey->ukopt.freeMarking = DEFAULT_CONF_FREEMARKING;
183 unikey->ukopt.macroEnabled = DEFAULT_CONF_MACROENABLED;
184
185 if (ibus_unikey_config_get_string(kConfig, CONFIG_SECTION, CONFIG_INPUTMETHOD, &str))
186 {
187 for (i = 0; i < NUM_INPUTMETHOD; i++)
188 {
189 if (!strncasecmp(str, Unikey_IMNames[i], 16))
190 {
191 unikey->im = Unikey_IM[i];
192 break;
193 }
194 }
195 } else {
196 // set default value
197 ibus_unikey_config_set_string(kConfig, CONFIG_SECTION, CONFIG_INPUTMETHOD, Unikey_IMNames[0]);
198 }
199 g_free(str);
200 str = NULL;
201
202 if (ibus_unikey_config_get_string(kConfig, CONFIG_SECTION, CONFIG_OUTPUTCHARSET, &str))
203 {
204 for (i = 0; i < NUM_OUTPUTCHARSET; i++)
205 {
206 if (!strncasecmp(str, Unikey_OCNames[i], 16))
207 {
208 unikey->oc = Unikey_OC[i];
209 break;
210 }
211 }
212 } else {
213 ibus_unikey_config_set_string(kConfig, CONFIG_SECTION, CONFIG_INPUTMETHOD, Unikey_OCNames[0]);
214 }
215 g_free(str);
216 str = NULL;
217
218 if (ibus_unikey_config_get_boolean(kConfig, CONFIG_SECTION, CONFIG_FREEMARKING, &b))
219 unikey->ukopt.freeMarking = b;
220 else {
221 ibus_unikey_config_set_boolean(kConfig, CONFIG_SECTION, CONFIG_FREEMARKING, DEFAULT_CONF_FREEMARKING);
222 }
223
224 if (ibus_unikey_config_get_boolean(kConfig, CONFIG_SECTION, CONFIG_MODERNSTYLE, &b))
225 unikey->ukopt.modernStyle = b;
226 else {
227 ibus_unikey_config_set_boolean(kConfig, CONFIG_SECTION, CONFIG_MODERNSTYLE, DEFAULT_CONF_MODERNSTYLE);
228 }
229
230 if (ibus_unikey_config_get_boolean(kConfig, CONFIG_SECTION, CONFIG_MACROENABLED, &b))
231 unikey->ukopt.macroEnabled = b;
232 else {
233 ibus_unikey_config_set_boolean(kConfig, CONFIG_SECTION, CONFIG_MACROENABLED, DEFAULT_CONF_MACROENABLED);
234 }
235
236 if (ibus_unikey_config_get_boolean(kConfig, CONFIG_SECTION, CONFIG_SPELLCHECK, &b))
237 unikey->ukopt.spellCheckEnabled = b;
238 else {
239 ibus_unikey_config_set_boolean(kConfig, CONFIG_SECTION, CONFIG_SPELLCHECK, DEFAULT_CONF_SPELLCHECK);
240 }
241
242 if (ibus_unikey_config_get_boolean(kConfig, CONFIG_SECTION, CONFIG_AUTORESTORENONVN, &b))
243 unikey->ukopt.autoNonVnRestore = b;
244 else {
245 ibus_unikey_config_set_boolean(kConfig, CONFIG_SECTION, CONFIG_AUTORESTORENONVN, DEFAULT_CONF_AUTONONVNRESTORE);
246 }
247
248 // load macro
249 gchar *fn = get_macro_file();
250 UnikeyLoadMacroTable(fn);
251 g_free(fn);
252
253 //update configures to UKEngine
254 UnikeySetInputMethod(unikey->im);
255 UnikeySetOutputCharset(unikey->oc);
256 UnikeySetOptions(&unikey->ukopt);
257}

259static void ibus_unikey_engine_property_activate(IBusEngine *engine,
260 const gchar *prop_name,
261 guint prop_state)
262{
263 IBusProperty *prop;
264 IBusText *label;
265 guint i, j;
266
267 kUnikey = (IBusUnikeyEngine *)engine;
268
269 // input method active
270 if (!strncmp(prop_name, CONFIG_INPUTMETHOD, strlen(CONFIG_INPUTMETHOD)))
271 {
272 for (i = 0; i < NUM_INPUTMETHOD; i++)
273 {
274 if (!strncmp(prop_name + strlen(CONFIG_INPUTMETHOD) + 1, Unikey_IMNames[i], 8))
275 {
276 kUnikey->im = Unikey_IM[i];
277
278 // update label
279 for (j = 0; j < kUnikey->prop_list->properties->len; j++)
280 {
281 prop = ibus_prop_list_get(kUnikey->prop_list, j);
282 if (!prop)
283 return;
284
285 if (!strncmp(ibus_property_get_key(prop), CONFIG_INPUTMETHOD, strlen(CONFIG_INPUTMETHOD)))
286 {
287 label = ibus_text_new_from_static_string(Unikey_IMNames[i]);
288 ibus_property_set_label(prop, label);
289 break;
290 }
291 } // end update label
292
293 // update property state
294 for (j = 0; j < kUnikey->menu_im->properties->len; j++)
295 {
296 prop = ibus_prop_list_get(kUnikey->menu_im, j);
297 if (!prop)
298 return;
299
300 if (!strncmp(ibus_property_get_key(prop), prop_name, 32))
301 ibus_property_set_state(prop, PROP_STATE_CHECKED);
302 else
303 ibus_property_set_state(prop, PROP_STATE_UNCHECKED);
304 } // end update property state
305
306 // update menu && save changes
307 ibus_engine_register_properties(engine, kUnikey->prop_list);
308 ibus_unikey_config_set_string(kConfig, CONFIG_SECTION, CONFIG_INPUTMETHOD, Unikey_IMNames[i]);
309 break;
310 }
311 }
312 } // end input method active
313
314 // output charset active
315 else if (!strncmp(prop_name, CONFIG_OUTPUTCHARSET, strlen(CONFIG_OUTPUTCHARSET)))
316 {
317 for (i = 0; i < NUM_OUTPUTCHARSET; i++)
318 {
319 if (!strncmp(prop_name + strlen(CONFIG_OUTPUTCHARSET) + 1, Unikey_OCNames[i], 16))
320 {
321 kUnikey->oc = Unikey_OC[i];
322
323 // update label
324 for (j = 0; j < kUnikey->prop_list->properties->len; j++)
325 {
326 prop = ibus_prop_list_get(kUnikey->prop_list, j);
327 if (!prop)
328 return;
329
330 if (!strncmp(ibus_property_get_key(prop), CONFIG_OUTPUTCHARSET, strlen(CONFIG_OUTPUTCHARSET)))
331 {
332 label = ibus_text_new_from_static_string(Unikey_OCNames[i]);
333 ibus_property_set_label(prop, label);
334 break;
335 }
336 } // end update label
337
338 // update property state
339 for (j = 0; j < kUnikey->menu_oc->properties->len; j++)
340 {
341 prop = ibus_prop_list_get(kUnikey->menu_oc, j);
342 if (!prop)
343 return;
344
345 if (!strncmp(ibus_property_get_key(prop), prop_name, 32) == 0)
346 ibus_property_set_state(prop, PROP_STATE_CHECKED);
347 else
348 ibus_property_set_state(prop, PROP_STATE_UNCHECKED);
349 } // end update property state
350
351 // update menu && save changes
352 ibus_engine_register_properties(engine, kUnikey->prop_list);
353 ibus_unikey_config_set_string(kConfig, CONFIG_SECTION, CONFIG_OUTPUTCHARSET, Unikey_OCNames[i]);
354 break;
355 }
356 }
357 } // end output charset active
358
359 // if Run setup {{{
360 else if (!strncmp(prop_name, "RunSetupGUI", 12))
361 {
362 pid_t pid = fork();
363 int errsv = errno;
364 int ret = 0;
365 switch (pid)
366 {
367 case -1:
368 g_warning("ibus_unikey_engine_property_activate: Cannot fork with errno %d\n", errsv);
369 exit(errsv);
370 return;
371 case 0:
372 ret = execl(LIBEXECDIR "/ibus-setup-unikey", "ibus-setup-unikey", NULL);
373 errsv = errno;
374
375 if (ret == -1)
376 {
377 g_warning("ibus_unikey_engine_property_activate: Cannot exec ibus-setup-unikey with errno %d\n", errsv);
378 }
379 exit(errsv);
380 default:
381 break;
382 }
383 } // }}} Run setup
384
385 UnikeySetInputMethod(kUnikey->im);
386 UnikeySetOutputCharset(kUnikey->oc);
387 UnikeySetOptions(&kUnikey->ukopt);
388
389 kParentClass->property_activate(engine, prop_name, prop_state);
390}

533static void ibus_unikey_engine_erase_chars(IBusEngine *engine, int num_chars)
534{
535 int k = num_chars;
536
537 guchar c;
538 int i;
539 for (i = kUnikey->preeditstr->length() - 1; i >= 0 && k > 0; i--)
540 {
541 c = kUnikey->preeditstr->at(i);
542
543 // count down if byte is begin byte of utf-8 char
544 if (c < (guchar)'\x80' || c >= (guchar)'\xC0')
545 {
546 k--;
547 }
548 }
549
550 kUnikey->preeditstr->erase(i+1);
551}

553static gboolean ibus_unikey_engine_process_key_event(IBusEngine *engine,
554 guint keysym,
555 guint keycode,
556 guint modifiers)
557{
558 kUnikey = (IBusUnikeyEngine *)engine;
559
560 static gboolean ok = ibus_unikey_engine_process_key_event_preedit(
561 engine, keysym, keycode, modifiers);
562
563 // check last keyevent with shift
564 if (keysym >= IBUS_space && keysym <=IBUS_asciitilde)
565 kUnikey->last_key_with_shift = modifiers & IBUS_SHIFT_MASK;
566 else
567 kUnikey->last_key_with_shift = FALSE;
568
569 return ok;
570}

572/** Use xev to get key state when debug
573 * Ref:
574 * https://developer.mozilla.org/en-US/docs/Web/API/KeyboardEvent/key/Key_Values
575 * https://wiki.archlinux.org/index.php/X_KeyBoard_extension
576 */
577static gboolean ibus_unikey_engine_process_key_event_preedit(IBusEngine *engine,
578 guint keyval,
579 guint keycode,
580 guint modifiers)
581{
582 // A key is released - not really important
583 if (modifiers & IBUS_RELEASE_MASK)
584 return FALSE;
585
586 // We only care about Control and Modifier 1
587 // (Usually Alt_L (0x40), Alt_R (0x6c), Meta_L (0xcd))
588 modifiers &= (IBUS_CONTROL_MASK | IBUS_MOD1_MASK);
589
590 // special key, hot key that we won't handle
591 if (modifiers & IBUS_CONTROL_MASK
592 || modifiers & IBUS_MOD1_MASK // alternate mask
593 || keyval == IBUS_Control_L
594 || keyval == IBUS_Control_R
595 || keyval == IBUS_Tab
596 || keyval == IBUS_Return
597 || keyval == IBUS_Delete
598 || keyval == IBUS_KP_Enter
599 || (IBUS_Home <= keyval && keyval <= IBUS_Insert)
600 || (IBUS_KP_Home <= keyval && keyval <= IBUS_KP_Delete)
601 )
602 {
603 ibus_unikey_engine_commit(engine);
604 return FALSE;
605 }
606
607 if ((IBUS_Caps_Lock <= keyval && keyval <= IBUS_Hyper_R)
608 || (!(modifiers & IBUS_SHIFT_MASK) && (keyval == IBUS_Shift_L || keyval == IBUS_Shift_R)) // when press one shift key
609 )
610 {
611 return FALSE;
612 }
613
614 if (IBUS_KP_Multiply <= keyval && keyval <= IBUS_KP_9)
615 {
616 ibus_unikey_engine_commit(engine);
617 return FALSE;
618 }
619
620 // capture BackSpace
621 if (keyval == IBUS_BackSpace)
622 {
623 UnikeyBackspacePress();
624
625 if (!UnikeyBackspaces || kUnikey->preeditstr->empty())
626 return FALSE;
627
628 if (kUnikey->preeditstr->length() <= (guint)UnikeyBackspaces)
629 {
630 kUnikey->preeditstr->clear();
631 ibus_engine_hide_preedit_text(engine);
632 }
633 else
634 {
635 ibus_unikey_engine_erase_chars(engine, UnikeyBackspaces);
636 ibus_unikey_engine_update_preedit_string(kUnikey);
637 }
638
639 // change tone position after press backspace
640 if (UnikeyBufChars > 0)
641 {
642 if (kUnikey->oc == CONV_CHARSET_XUTF8)
643 {
644 kUnikey->preeditstr->append((gchar *)UnikeyBuf, UnikeyBufChars);
645 }
646 else
647 {
648 static guchar buf[CONVERT_BUF_SIZE];
649 int bufSize = CONVERT_BUF_SIZE;
650
651 latinToUtf(buf, UnikeyBuf, UnikeyBufChars, &bufSize);
652 kUnikey->preeditstr->append((gchar *)buf, CONVERT_BUF_SIZE - bufSize);
653 }
654
655 ibus_unikey_engine_update_preedit_string(kUnikey);
656 }
657 return TRUE;
658 } // end capture BackSpace
659
660 // capture ascii printable char
661 if ((keyval >= IBUS_space && keyval <= IBUS_asciitilde)
662 || keyval == IBUS_Shift_L || keyval == IBUS_Shift_R) // sure this have IBUS_SHIFT_MASK
663 {
664
665 UnikeySetCapsState(modifiers & IBUS_SHIFT_MASK, modifiers & IBUS_LOCK_MASK);
666
667 // shift + space, shift + shift event
668 if ((kUnikey->last_key_with_shift == FALSE && (modifiers & IBUS_SHIFT_MASK)
669 && keyval == IBUS_space && !UnikeyAtWordBeginning())
670 || (keyval == IBUS_Shift_L || keyval == IBUS_Shift_R) // sure this have IBUS_SHIFT_MASK
671 )
672 {
673 UnikeyRestoreKeyStrokes();
674 } // end shift + space, shift + shift event
675
676 else
677 {
678 UnikeyFilter(keyval);
679 }
680 // end process keyval
681
682 // process result of ukengine
683 if (UnikeyBackspaces > 0)
684 {
685 if (kUnikey->preeditstr->length() <= (guint)UnikeyBackspaces)
686 {
687 kUnikey->preeditstr->clear();
688 }
689 else
690 {
691 ibus_unikey_engine_erase_chars(engine, UnikeyBackspaces);
692 }
693 }
694
695 if (UnikeyBufChars > 0)
696 {
697 if (kUnikey->oc == CONV_CHARSET_XUTF8)
698 {
699 kUnikey->preeditstr->append((gchar *)UnikeyBuf, UnikeyBufChars);
700 }
701 else
702 {
703 static unsigned char buf[CONVERT_BUF_SIZE];
704 int bufSize = CONVERT_BUF_SIZE;
705
706 latinToUtf(buf, UnikeyBuf, UnikeyBufChars, &bufSize);
707 kUnikey->preeditstr->append((gchar *)buf, CONVERT_BUF_SIZE - bufSize);
708 }
709 }
710 else if (keyval != IBUS_Shift_L && keyval != IBUS_Shift_R) // if ukengine not process
711 {
712 static char s[6];
713 static int n = g_unichar_to_utf8(keyval, s); // convert ucs4 to utf8 char
714 kUnikey->preeditstr->append(s, n);
715 }
716 // end process result of ukengine
717
718 // commit string: if need
719 if (kUnikey->preeditstr->length() > 0)
720 {
721 guchar last_char = kUnikey->preeditstr->at(kUnikey->preeditstr->length() - 1);
722 static guint i;
723 for (i = 0; i < sizeof(WordBreakSyms); i++)
724 {
725 if (WordBreakSyms[i] == last_char && WordBreakSyms[i] == keyval)
726 {
727 ibus_unikey_engine_commit(engine);
728 return TRUE;
729 }
730 }
731 }
732 // end commit string
733
734 ibus_unikey_engine_update_preedit_string(kUnikey);
735 return TRUE;
736 } //end capture printable char
737
738 // non process key
739 ibus_unikey_engine_commit(engine);
740 return FALSE;
741}

743static void ibus_unikey_engine_clean_buffer(IBusEngine *engine)
744{
745 UnikeyResetBuf();
746 kUnikey->preeditstr->clear();
747 ibus_engine_hide_preedit_text(engine);
748}

750static void ibus_unikey_engine_commit(IBusEngine *engine)
751{
752 kUnikey = (IBusUnikeyEngine *)engine;
753
754 if (kUnikey->preeditstr->length() > 0)
755 {
756 IBusText *text;
757
758 text = ibus_text_new_from_static_string(kUnikey->preeditstr->c_str());
759 ibus_engine_commit_text(engine, text);
760 }
761
762 ibus_unikey_engine_clean_buffer(engine);
763}
