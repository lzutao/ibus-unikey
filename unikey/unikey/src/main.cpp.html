33static void
34start_component(void)
35{
36 ibus_init();
37 //!< An IBusBus connects with IBus daemon.
38 IBusBus *bus = ibus_bus_new();
39 if (!ibus_bus_is_connected(bus)) {
40 g_warning("start_component: Cannot connect to IBus daemon.\n");
41 exit(0);
42 }
43
44 g_signal_connect(bus, "disconnected", G_CALLBACK(ibus_disconnected_cb), NULL);
45
46 //!< Factory for creating engine instances.
47 IBusFactory *factory = ibus_factory_new(ibus_bus_get_connection(bus));
48
49 IBusComponent *component = ibus_unikey_get_component();
50
51 factory = ibus_factory_new(ibus_bus_get_connection(bus));
52
53 GList *engines = ibus_component_get_engines(component);
54 GList *p = NULL;
55 const gchar *engine_name = NULL;
56
57 for (p = engines; p != NULL; p = p->next)
58 {
59 IBusEngineDesc *engine = (IBusEngineDesc *)p->data;
60 engine_name = ibus_engine_desc_get_name(engine);
61 ibus_factory_add_engine(factory, engine_name, IBUS_TYPE_UNIKEY_ENGINE);
62 }
63
64 if (kIBus)
65 {
66 if (!ibus_bus_request_name(bus, UNIKEY_DBUS_NAME, 0))
67 {
68 g_debug("start_component: Cannot request " UNIKEY_DBUS_NAME " from IBus daemon synchronously.\n");
69 goto fail;
70 }
71 }
72 else
73 {
74 if (!ibus_bus_register_component(bus, component))
75 {
76 g_debug("start_component: Cannot register Unikey component to IBus daemon synchronously.\n");
77 goto fail;
78 }
79 }
80
81 goto check_ok;
82fail:
83 g_object_unref(component);
84 exit(EXIT_FAILURE);
85check_ok:
86 g_object_unref(component);
87
88 ibus_unikey_init(bus);
89 ibus_main();
90 ibus_unikey_exit();
91 g_object_unref(factory);
92 g_object_unref(bus);
93}

95int
96main(gint argc, gchar **argv)
97{
98 // select a locale based on the user choice of the appropriate environment variables
99 setlocale(LC_ALL, "");
100 bindtextdomain(GETTEXT_PACKAGE, LOCALEDIR);
101 textdomain(GETTEXT_PACKAGE);
102
103 // Parse the command line
104 GOptionContext* context = g_option_context_new("- ibus unikey engine component");
105 g_option_context_add_main_entries(context, kEntries, "ibus-unikey");
106
107 GError *error = NULL;
108 gboolean parse_ok = g_option_context_parse(context, &argc, &argv, &error);
109 g_option_context_free(context);
110 if (!parse_ok) {
111 g_print("Option parsing failed: %s\n", error->message);
112 g_error_free(error);
113 exit(EXIT_FAILURE);
114 }
115
116 if (kVersion)
117 {
118 g_print(PACKAGE_STRING " (engine component)\n"
119 " Copyright (C) 2009 - 2012 Ubuntu-VN <http://www.ubuntu-vn.org>\n"
120 " Author: Lê Quốc Tuấn <mr.lequoctuan@gmail.com>\n"
121 " Homepage: <http://ibus-unikey.googlecode.com>\n"
122 " License: GNU GPLv3\n"
123 );
124 exit(EXIT_SUCCESS);
125 }
126
127 start_component();
128
129 return 0;
130}
