279/**
280 * Add unchecked fixture setup/teardown functions to a test case
281 *
282 * Unchecked fixture functions are run at the start and end of the
283 * test case, and not before and after unit tests. Further,
284 * unchecked fixture functions are not run in a separate address space,
285 * like test functions, and so must not exit or signal (e.g.,
286 * segfault).
287 *
288 * Also, when run in CK_NOFORK mode, unchecked fixture functions may
289 * lead to different unit test behavior if unit tests change data
290 * setup by the fixture functions.
291 *
292 * Note that if a setup function fails, the remaining setup functions
293 * will be omitted, as will the test case and the teardown functions.
294 * If a teardown function fails the remaining teardown functins will be
295 * omitted.
296 *
297 * @param tc test case to add unchecked fixture setup/teardown to
298 * @param setup function to add to be executed before the test case;
299 * if NULL no setup function is added
300 * @param teardown function to add to be executed after the test case;
301 * if NULL no teardown function is added
302 * @since 0.8.0
303 */
304CK_DLL_EXP void CK_EXPORT tcase_add_unchecked_fixture(TCase * tc, SFun setup,
305 SFun teardown);
306
307/**
308 * Add checked fixture setup/teardown functions to a test case
309 *
310 * Checked fixture functions are run before and after each unit test inside
311 * of the address space of the test. Thus, if using CK_FORK
312 * mode the separate process running the unit test will survive signals
313 * or unexpected exits in the fixture function. Also, if the setup
314 * function is idempotent, unit test behavior will be the same in
315 * CK_FORK and CK_NOFORK modes.
316 *
317 * However, since fixture functions are run before and after each unit
318 * test, they should not be expensive code.
319 *
320 * Note that if a setup function fails, the remaining setup functions
321 * will be omitted, as will the test and the teardown functions. If a
322 * teardown function fails the remaining teardown functins will be
323 * omitted.
324 *
325 * @param tc test case to add checked fixture setup/teardown to
326 * @param setup function to add to be executed before each unit test in
327 * the test case; if NULL no setup function is added
328 * @param teardown function to add to be executed after each unit test in
329 * the test case; if NULL no teardown function is added
330 *
331 * @since 0.8.0
332*/
333CK_DLL_EXP void CK_EXPORT tcase_add_checked_fixture(TCase * tc, SFun setup,
334 SFun teardown);

336/**
337 * Set the timeout for all tests in a test case.
338 *
339 * A test that lasts longer than the timeout (in seconds) will be killed
340 * and thus fail with an error.
341 *
342 * If not set, the default timeout is one assigned at compile time. If
343 * the environment variable CK_DEFAULT_TIMEOUT is defined and no timeout
344 * is set, the value in the environment variable is used.
345 *
346 * If Check is compile without fork() support this call is ignored,
347 * as timeouts are not possible.
348 *
349 * @param tc test case to assign timeout to
350 * @param timeout to use, in seconds. If the value contains a decimal
351 * portion, but no high resolution timer is available,
352 * the value is rounded up to the nearest second.
353 *
354 * @since 0.9.2
355 */
356CK_DLL_EXP void CK_EXPORT tcase_set_timeout(TCase * tc, double timeout);
